          const pr3 = new Promise((_, rej) => {
            console.log('will reject in 3 s')
            setTimeout(() => { rej() }, 3000)
          })
          Promise.all([pr, pr2, pr3]).then(() => {
            console.log('all completed')
          }).catch(() => {
            console.error('promise all failed')
          })
          Promise.allSettled([pr, pr2, pr3]).then('all have settled, despite failing')
